AngularBreakpoint

Add responsive design awareness to your angular templates

Template

<section ng-app="myApp" breakpoint>
  <div ng-class="{ 'mobile': breakpoint.xs || breakpoint.sm, 'tablet': breakpoint.md, 'desktop': breakpoint.lg }">
    <article ng-if="breakpoint.xs || breakpoint.sm">
      Small Screens
    </article>
    <article ng-if="breakpoint.md">
      Medium Screens
    </article>
    <article ng-if="breakpoint.lg">
      Large Screens
    </article>
  </div>
</section>

CSS

.mobile {
  background: tomato;
}
.tablet {
  background: gold;
}
.desktop {
  background: mediumturquoise;
}

JavaScript

angular.module('myApp', ['ngBreakpoint']);
